## Motivation

With subdued demand on the stablecoin leg, the previous platform fees plus a 1% annual volatile-leg fee created unnecessary drag. We lower friction by:

- Reducing the annual volatile-leg fee to **0.75%** (charged weekly on-chain).
- Reducing **platform** fees on primary-market operations (mint/redeem).
- Keeping the **DApp** fee at **0.10%** (unchanged), preserving app-level sustainability.

**Contract:** `FeesAndBitprorateProposal.sol`  
**Safety:** Delegatecall-safe; `execute()` does not write changer storage.


- **MoC V1:**
  - `setBitProRate(bitProRate)`

  - `setTCInterestRate(bitProRate)`
  - For each `(key, value)`: call the corresponding setter (including `setFeeTokenPct`)

## Appendix B â€” Deploy Script (TypeScript)

`scripts/deploy-fees-and-bitprorate.ts`:

```ts
import { ethers } from "hardhat";
import fs from "node:fs";
import { resolve } from "node:path";

const RocV2FeeKeyMap: Record<string, number> = {
  TcMintFee: 0,
  TcRedeemFee: 1,
  SwapTPforTPFee: 2,
  SwapTPforTCFee: 3,
  SwapTCforTPFee: 4,
  RedeemTCandTPFee: 5,
  MintTCandTPFee: 6,
  FeeTokenPct: 7,
};

const TX_NAME_TO_ID: Record<string, number> = {
  MINT_BPRO_FEES_RBTC: 1,
  REDEEM_BPRO_FEES_RBTC: 2,
  MINT_DOC_FEES_RBTC: 3,
  REDEEM_DOC_FEES_RBTC: 4,
  MINT_BTCX_FEES_RBTC: 5,
  REDEEM_BTCX_FEES_RBTC: 6,
  MINT_BPRO_FEES_MOC: 7,
  REDEEM_BPRO_FEES_MOC: 8,
  MINT_DOC_FEES_MOC: 9,
  REDEEM_DOC_FEES_MOC: 10,
  MINT_BTCX_FEES_MOC: 11,
  REDEEM_BTCX_FEES_MOC: 12,
};

function toWei18(x: string): bigint {
  // decimal string "0.0015" -> 18 decimals
  const [i, f = ""] = x.split(".");
  const frac = (f + "0".repeat(18)).slice(0, 18);
  return BigInt(`${i}${frac}`);
}

async function main() {
  const cfgPath = resolve(process.env.CONFIG || "deployConfig-rskMainnet.json");
  const cfg = JSON.parse(fs.readFileSync(cfgPath, "utf8"));

  const mocInrate: string = cfg.MoCInrate || cfg.mocInrate;
  const rocV2: string = cfg.RocV2 || cfg.rocV2;
  if (!mocInrate || !rocV2) throw new Error("Missing MoCInrate/RocV2 in config");

  // BitPro weekly rate (decimal string -> 18d)
  const bitProRate = toWei18(cfg.bitProRate);

  // MoC V1 commissions (object -> array of structs)
  const commissionRates = Object.entries<string>(cfg.commissionRates || {})
    .map(([name, dec]) => {
      const txType = TX_NAME_TO_ID[name];
      if (!txType) throw new Error(`Unknown txType key: ${name}`);
      return { txType, fee: toWei18(dec) };
    })
    .sort((a, b) => a.txType - b.txType);

  // ROC V2 fees (object -> array of structs)
  const rocV2Fees = Object.entries<string>(cfg.rocV2Fees || {})
    .map(([key, dec]) => {
      const id = RocV2FeeKeyMap[key];
      if (id === undefined) throw new Error(`Unknown ROC V2 key: ${key}`);
      return { key: id, value: toWei18(dec) };
    })
    .sort((a, b) => a.key - b.key);

  const Factory = await ethers.getContractFactory("FeesAndBitprorateProposal");
  const proposal = await Factory.deploy(mocInrate, rocV2, bitProRate, commissionRates, rocV2Fees);
  await proposal.waitForDeployment();

  console.log("Deployed FeesAndBitprorateProposal at:", await proposal.getAddress());
  console.log("MoCInrate:", mocInrate);
  console.log("ROC V2   :", rocV2);
  console.log("bitProRate (wei):", bitProRate.toString());
  console.log("commissionRates:", commissionRates);
  console.log("rocV2Fees:", rocV2Fees);
}

main().catch((e) => {
  console.error(e);
  process.exit(1);
});
```
